Linear Regression

-- Create and train a linear regression model
model := create {LINEAR_REGRESSION_MODEL}.make
    .set_learning_rate (0.01)
    .set_max_iterations (100)

model.train (training_X, training_y)

-- Make predictions
prediction := model.predict (<< 1.5, 2.5 >>)

Logistic Regression

-- Binary and multiclass classification
model := create {LOGISTIC_REGRESSION_MODEL}.make
    .set_learning_rate (0.01)
    .set_max_iterations (100)

model.train (training_X, training_y)

-- Predict class
class_label := model.predict (<< 1.5, 2.5 >>)

-- Get probability
probability := model.predict_proba (<< 1.5, 2.5 >>)

Decision Tree

-- Tree-based classification
model := create {DECISION_TREE_CLASSIFIER}.make
    .set_max_depth (5)
    .set_min_samples_split (2)

model.train (training_X, training_y)

-- Predict class
class_label := model.predict (<< 1.5, 2.5 >>)

Random Forest

-- Ensemble of decision trees
model := create {RANDOM_FOREST_CLASSIFIER}.make
    .set_num_trees (100)
    .set_max_depth (10)

model.train (training_X, training_y)

-- Predict class and probabilities
class_label := model.predict (<< 1.5, 2.5 >>)
probabilities := model.predict_proba (<< 1.5, 2.5 >>)

Support Vector Machine

-- Linear SVM for classification
model := create {SVM_LINEAR}.make
    .set_c_param (1.0)
    .set_tolerance (0.0001)

model.train (training_X, training_y)

-- Predict and get decision value
class_label := model.predict (<< 1.5, 2.5 >>)
decision := model.decision_function (<< 1.5, 2.5 >>)

Neural Network

-- Multi-layer perceptron
model := create {NEURAL_NETWORK_CLASSIFIER}.make
    .set_hidden_layers (<< 32, 16 >>)
    .set_learning_rate (0.01)
    .set_max_iterations (200)

model.train (training_X, training_y)

-- Predict and get probabilities
class_label := model.predict (<< 1.5, 2.5 >>)
probabilities := model.predict_proba (<< 1.5, 2.5 >>)

Common Operations

Fluent Configuration

All configuration methods return like Current for method chaining:

model := create {LINEAR_REGRESSION_MODEL}.make
    .set_learning_rate (0.001)
    .set_max_iterations (500)

Training Data Format

training_X is an ARRAY [ARRAY [REAL_64]] (rows of features)

training_y is ARRAY [REAL_64] for regression or ARRAY [INTEGER] for classification

Model Queries

Check training status and learned parameters:

if model.is_trained then
    features := model.features_learned  -- for regression
    classes := model.classes_learned    -- for classification
end

Error Handling

Use preconditions to validate inputs:

require
    x_valid: a_x /= Void and a_x.count > 0
    y_valid: a_y /= Void and a_y.count > 0
    same_length: a_x.count = a_y.count
